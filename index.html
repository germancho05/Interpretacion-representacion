    // --- SCORM 1.2 Integration ---

    const scorm = (function() {

        let API = null;

        let connectionActive = false;


        function findAPI(win) {

            let findAPITries = 0;

            // Wrap in a try-catch block to prevent cross-origin errors

            try {

                while ((win.API == null) && (win.parent != null) && (win.parent != win)) {

                    findAPITries++;

                    if (findAPITries > 7) {

                        console.warn("Error: Could not find SCORM API. Exceeded max tries.");

                        return null;

                    }

                    win = win.parent;

                }

            } catch (e) {

                console.warn("SCORM findAPI failed with error:", e);

            }

            return win.API;

        }


        function init() {

            let win = window;

            // Wrap in a try-catch block to prevent cross-origin errors

            try {

                API = findAPI(win);

                if ((API == null) && (win.opener != null) && (typeof(win.opener) != "undefined")) {

                    API = findAPI(win.opener);

                }

            } catch(e) {

                console.warn("SCORM init failed with error:", e);

            }


            if (API == null) {

                console.warn("SCORM API not found. Content will run in standalone mode.");

                return;

            }


            connectionActive = API.LMSInitialize("");

            if(connectionActive.toString() === "true"){

                const status = API.LMSGetValue("cmi.core.lesson_status");

                if(status === "not attempted"){

                    API.LMSSetValue("cmi.core.lesson_status", "incomplete");

                }

                API.LMSCommit("");

            }

        }


        function setScore(newScore) {

            if (connectionActive.toString() === "true") {

                API.LMSSetValue("cmi.core.score.raw", newScore);

                API.LMSCommit("");

            } else {

                console.log(`Standalone Mode: Score set to ${newScore}`);

            }

        }


        function setCompleted() {

            if (connectionActive.toString() === "true") {

                API.LMSSetValue("cmi.core.lesson_status", "completed");

                API.LMSCommit("");

            } else {

                console.log("Standalone Mode: Course completed.");

            }

        }

        

        function finish() {

            if (connectionActive.toString() === "true") {

                API.LMSFinish("");

                connectionActive = false;

            }

        }


        return { init, setScore, setCompleted, finish };

    })();

                span.classList.add('selectable');

                span.dataset.id = index;

                span.onclick = () => highlight(span);

            }

            problemContainer.appendChild(span);

        });

        solutionDiv.classList.add('hidden');

        feedbackDiv.innerHTML = '';

        feedbackDiv.classList.add('hidden');

        document.getElementById('retry-cap2').classList.add('hidden');

    }

    document.querySelectorAll('.tool-button').forEach(button => button.addEventListener('click', () => { currentTool = button.dataset.tool; }));

    function highlight(span) {

        span.className = 'selectable'; // Reset classes

        span.classList.add(`highlight-${currentTool}`);

        span.dataset.userSelection = currentTool;

    }

    function checkDetectiveWork() {

        let correctCount = 0;

        const selectableSpans = problemContainer.querySelectorAll('.selectable');

        selectableSpans.forEach(span => {

            if (problemData[span.dataset.id].type === span.dataset.userSelection) correctCount++;

        });

        feedbackDiv.classList.remove('hidden');

        if (correctCount === selectableSpans.length) {

            feedbackDiv.className = 'mt-4 p-3 rounded-lg bg-green-100 text-green-800';

            feedbackDiv.innerHTML = `<strong>¡Excelente trabajo, detective!</strong> (+10 Puntos) Has identificado todas las pistas correctamente.`;

            if (!cap2Completed) { score += 10; cap2Completed = true; updateScoreAndLMS(); }

            solutionDiv.classList.remove('hidden');

        } else {

            feedbackDiv.className = 'mt-4 p-3 rounded-lg bg-red-100 text-red-800';

            feedbackDiv.innerHTML = `<strong>Casi lo logras.</strong> Tienes ${correctCount} de ${selectableSpans.length} pistas correctas. ¡Inténtalo de nuevo!`;

            document.getElementById('retry-cap2').classList.remove('hidden');

        }

    }
